CREATE DATABASE GIVANILDO;

CREATE TABLE TB_CLIENTE (
    CLI_ID INT AUTO_INCREMENT PRIMARY KEY,
    CLI_LOGIN VARCHAR(250) NOT NULL,
    CLI_SENHA VARCHAR(250) NOT NULL UNIQUE
);

CREATE TABLE TB_CATEGORIA (
    CATE_ID INT PRIMARY KEY,
    CATE_DESCRICAO VARCHAR(250)
);

CREATE TABLE TB_TAREFA (
    TAREFA_ID INT AUTO_INCREMENT PRIMARY KEY,
    TAREFA_TITULO VARCHAR(100),
    TAREFA_DESCRICAO VARCHAR(250),
    TAREFA_DATA_INICIO DATETIME,
    TAREFA_DATA_FIM DATETIME,
    TAREFA_STATUS CHAR(1),
    TAREFA_FINALIZADA TINYINT(1),
    TAREFA_CLIENTE INT,
    TAREFA_CATEGORIA INT,
    CONSTRAINT FK_CLIENTE_TAREFA FOREIGN KEY (TAREFA_CLIENTE) REFERENCES TB_CLIENTE(CLI_ID) ON
    DELETE RESTRICT,
    CONSTRAINT FK_CATEGORIA_TAREFA FOREIGN KEY (TAREFA_CATEGORIA) REFERENCES TB_CATEGORIA            (CATE_ID) ON DELETE SET NULL
);
